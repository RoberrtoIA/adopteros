CREATE TABLE contactos (
  id INT PRIMARY KEY,
  telefono_upfoter VARCHAR(255) NOT NULL,
  email VARCHAR(255) NOT NULL,
  instagram_link VARCHAR(255) NOT NULL,
  twitter_link VARCHAR(255) NOT NULL,
  facebook_link VARCHAR(255) NOT NULL
);

DELIMITER //
CREATE PROCEDURE `contacto_update`(
	IN _telefono VARCHAR(255), 
	IN _email VARCHAR(255),
	IN _instagram_link VARCHAR(255),
	IN _twitter_link VARCHAR(255),
	IN _facebook_link VARCHAR(255)
)
BEGIN  
    UPDATE contactos
    SET 
	 telefono_upfoter = _telefono, 
    email = _email,
    instagram_link = _instagram_link,
    twitter_link = _twitter_link,
    facebook_link = _facebook_link
    WHERE id = 1;
END//
DELIMITER ;
